Example

Here is a simple Java class:

import java.util.*;

public class test {
    public static void main(String[] args) {
	Collection c = Collections.singleton(new Object());
    }
}

Here are its dependencies as an XML document:

<dependencies>
    <package>
        <name></name>
        <class>
            <name>test</name>
            <outbound type="class">java.lang.Object</outbound>
            <feature>
                <name>test.main(java.lang.String[])</name>
                <outbound type="class">java.lang.Object</outbound>
                <outbound type="feature">java.lang.Object.Object()</outbound>
                <outbound type="class">java.lang.String</outbound>
                <outbound type="feature">java.util.Collections.singleton(java.lang.Object)</outbound>
            </feature>
            <feature>
                <name>test.test()</name>
                <outbound type="feature">java.lang.Object.Object()</outbound>
            </feature>
        </class>
    </package>
    <package>
        <name>java.lang</name>
        <class>
            <name>java.lang.Object</name>
            <inbound type="class">test</inbound>
            <inbound type="feature">test.main(java.lang.String[])</inbound>
            <feature>
                <name>java.lang.Object.Object()</name>
                <inbound type="feature">test.main(java.lang.String[])</inbound>
                <inbound type="feature">test.test()</inbound>
            </feature>
        </class>
        <class>
            <name>java.lang.String</name>
            <inbound type="feature">test.main(java.lang.String[])</inbound>
        </class>
    </package>
    <package>
        <name>java.util</name>
        <class>
            <name>java.util.Collections</name>
            <feature>
                <name>java.util.Collections.singleton(java.lang.Object)</name>
                <inbound type="feature">test.main(java.lang.String[])</inbound>
            </feature>
        </class>
    </package>
</dependencies>
